    # 1、依赖库打包

    在终端输入./copy.sh，将所需库文件拷贝到exe可执行程序的文件夹下

    exe="libmk_api.so" #发布的程序名称
des="/home/user/xxxx" #程序路径

deplist=$(ldd $exe | awk  '{if (match($3,"/")){ printf("%s "),$3 } }')
cp $deplist $des

    # 2、程序运行

    ​ 在终端输入./xxx.sh即可运行，其中**.sh文件命名必须与可执行程序名字一样**，如执行文件名test， .sh 文件名为test.sh

    #!/bin/sh
appname=`basename $0 | sed s,\.sh$,,`
 
dirname=`dirname $0`
tmp="${dirname#?}"
 
if [ "${dirname%$tmp}" != "/" ]; then
dirname=$PWD/$dirname
fi
LD_LIBRARY_PATH=$dirname
export LD_LIBRARY_PATH
$dirname/$appname "$@"

# 3.在/etc/rc.d/rc.local末尾添加如下内容：

    # 5、环境变量

    终端执行，临时有效。

    export PATH=$PATH:/home/user/xxxx/
    1

    • /etc/profile

      • 一般用于设置所有用户使用的全局变量，当用户登录时，该文件被执行一次；

    • /etc/bashrc

      • 当终端被打开时，该文件被读取。即，每次新打开一个终端，该文件就会被读取；

    • ~/.bash_profile和~/.profile

      • 只对单个用户生效，当用户登录时该文件仅执行一次。不同的Linux下文件名称不一样，可能是 ~/.bash_profile， ~/.bash_login 或 ~/.profile 其中的一种或几种，如果存在几种的话，那么执行的顺序便是：~/.bash_profile、 ~/.bash_login、~/.profile。比如 Ubuntu 系统一般是 ~/.profile 文件。

    • ~/.bashrc

      • 只对单个用户生效，当登录以及每次打开新的终端时，该文件被读取；

    • /etc/environment

      • 设置整个系统的环境，而/etc/profile是设置所有用户的环境，前者与登录用户无关，后者与登录用户有关。系统是先执行/etc/environment，再执行/etc/profile。
    PATH="$PATH:/home/user/xxxx/"
export PATH

source /etc/profile
