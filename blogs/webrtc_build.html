    # 4、一些环境设置

    set vs2019_install=D:\Program Files (x86)\Microsoft Visual Studio\2019\Professional
set GYP_MSVS_OVERRIDE_PATH=D:\Program Files (x86)\Microsoft Visual Studio\2019\Professional
set GYP_GENERATORS=msvs-ninja,ninja
set DEPOT_TOOLS_WIN_TOOLCHAIN=0
set WINDOWSSDKDIR=D:\Windows Kits\10

    # 5、设置科学上网端口代理

    //注意https_proxy的http是不带s的
set http_proxy=http://127.0.0.1:8889
set https_proxy=http://127.0.0.1:8889

//可能还需要设置git的代理
git config --global http.proxy 'http://127.0.0.1:8889'
git config --global https.proxy 'https://127.0.0.1:8889'

取消代理：git config --global --unset http.proxy

    # 7、下载webrtc

    mkdir webrtc-checkout
cd webrtc-checkout
fetch --nohooks webrtc
gclient sync
cd src
git checkout -b m84 refs/remotes/branch-heads/4147
gclient sync
